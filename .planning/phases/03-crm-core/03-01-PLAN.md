<!-- Schema from postgres/init/001_schema.sql -->

patients table:
  id UUID PK DEFAULT gen_random_uuid()
  first_name TEXT NOT NULL
  last_name TEXT NOT NULL
  phone TEXT NOT NULL
  phone_normalized TEXT NOT NULL (UNIQUE constraint: uq_patients_phone_normalized)
  email TEXT
  notes TEXT
  consent_status BOOLEAN DEFAULT false
  source TEXT DEFAULT 'csv_import'
  created_at TIMESTAMPTZ DEFAULT now()
  updated_at TIMESTAMPTZ DEFAULT now()
  Indexes: gin_trgm on (first_name || ' ' || last_name), btree on phone_normalized, btree on created_at

tags table:
  id UUID PK DEFAULT gen_random_uuid()
  name TEXT NOT NULL UNIQUE
  color TEXT DEFAULT '#6366f1'
  created_at TIMESTAMPTZ DEFAULT now()

patient_tags table:
  patient_id UUID FK -> patients(id) ON DELETE CASCADE
  tag_id UUID FK -> tags(id) ON DELETE CASCADE
  assigned_at TIMESTAMPTZ DEFAULT now()
  PK (patient_id, tag_id)

message_templates table:
  id UUID PK DEFAULT gen_random_uuid()
  name TEXT NOT NULL UNIQUE
  body TEXT NOT NULL
  variables TEXT[] DEFAULT '{}'
  category TEXT DEFAULT 'general'
  is_active BOOLEAN DEFAULT true
  created_at TIMESTAMPTZ DEFAULT now()
  updated_at TIMESTAMPTZ DEFAULT now()
